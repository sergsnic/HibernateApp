create table Person
(
    id int primary key generated by default as identity,
    name varchar(100) not null,
    age int check ( age < 100 )
);
create table Item
(
    id int primary key generated by default as identity,
    person_id int references Person(id) on DELETE set null,
    item_name varchar(100) not null
);
